    "def make_time_array(t0, tf, dt_sec):\n",
    "    \"\"\"Create a TDB EpochArray from t0 to tf, including the final epoch.\"\"\"\n",
    "    return EpochArray(np.arange(t0, tf + dt_sec, dt_sec), sys=\"TDB\")\n",
    "\n",
    "\n",
    "def print_arc(name, t_start, t_end):\n",
    "    \"\"\"Print an arc window in UTC and hours.\"\"\"\n",
    "    try:\n",
    "        t0_str = SpiceManager.et2utc(t_start, \"C\", 3)\n",
    "        tf_str = SpiceManager.et2utc(t_end, \"C\", 3)\n",
    "    except Exception:\n",
    "        t0_str = f\"{t_start:.3f} ET\"\n",
    "        tf_str = f\"{t_end:.3f} ET\"\n",
    "\n",
    "    duration_hr = (t_end - t_start) / 3600.0\n",
    "    print(f\"{name}\")\n",
    "    print(f\"  Start   : {t0_str}\")\n",
    "    print(f\"  End     : {tf_str}\")\n",
    "    print(f\"  Duration: {duration_hr:.3f} hr\")\n",
    "\n",
    "\n",
    "def empty_pv_state(epoch, spacecraft, origin, frame):\n",
    "    \"\"\"Create an empty position/velocity state used by sequence arcs after the first arc.\"\"\"\n",
    "    return StateArray(\n",
    "        epoch=EpochArray(epoch, sys=\"TDB\"),\n",
    "        origin=origin,\n",
    "        state=StateDefinition.empty_pv(spacecraft, frame=frame),\n",
    "    )\n",
    "\n",
    "\n",
    "def plot_mass_history(mission, spacecraft, title=\"Mass Evolution\"):\n",
    "    \"\"\"Plot spacecraft mass along the propagated mission sequence.\"\"\"\n",
    "    epochs = mission.total_epochsTDB.times.values\n",

    "# The exact trajectory attribute names may differ depending on the current Scarabaeus branch.\n",
    "# This helper tries common patterns and raises a useful error if the branch API changed.\n",
    "def get_cartesian_history(mission):\n",
    "    candidates = [\n",
    "        \"total_state\",\n",
    "        \"total_states\",\n",
    "        \"state\",\n",
    "        \"states\",\n",
    "        \"trajectory\",\n",
    "    ]\n",
    "    for attr in candidates:\n",
    "        if hasattr(mission, attr):\n",
    "            obj = getattr(mission, attr)\n",
    "            try:\n",
    "                vals = obj.values\n",
    "            except Exception:\n",
    "                vals = obj\n",
    "            arr = np.asarray(vals)\n",
    "            if arr.ndim == 2 and arr.shape[1] >= 6:\n",
    "                return arr[:, :6]\n",
    "            if arr.ndim == 2 and arr.shape[0] >= 6:\n",
    "                return arr[:6, :].T\n",
    "    raise AttributeError(\n",
    "        \"Could not infer the Cartesian state history from this mission object. \"\n",
    "        \"Inspect dir(mission) and update get_cartesian_history for this branch.\"\n",
    "    )\n",
    "\n",
    "try:\n",
