    "# B-plane Targeting Demo: OSIRIS-REx 2017 Earth Gravity Assist\n",
    "\n",

    "\n",

    "# ============================================================\n",
    "# Fancy custom B-plane covariance plot\n",
    "# ============================================================\n",
    "\n",
    "import numpy as np\n",
    "import matplotlib.pyplot as plt\n",
    "from matplotlib.patches import Ellipse\n",
    "\n",
    "# Extract covariance matrix\n",
    "P = P_bplane[:2, :2].values\n",
    "\n",
    "# Target coordinates\n",
    "bt = float(BplaneObj._bt_ref.values)\n",
    "br = float(BplaneObj._br_ref.values)\n",
    "\n",
    "# Eigen decomposition\n",
    "eigvals, eigvecs = np.linalg.eigh(P)\n",
    "\n",
    "# Sort largest first\n",
    "order = eigvals.argsort()[::-1]\n",
    "eigvals = eigvals[order]\n",
    "eigvecs = eigvecs[:, order]\n",
    "\n",
    "# Ellipse angle\n",
    "angle = np.degrees(np.arctan2(eigvecs[1, 0], eigvecs[0, 0]))\n",
    "\n",
    "# ============================================================\n",
    "# Plot\n",
    "# ============================================================\n",
    "\n",
    "fig, ax = plt.subplots(figsize=(9, 9))\n",
    "\n",
    "# Background style\n",
    "ax.set_facecolor(\"#fafafa\")\n",
    "\n",
    "# Sigma ellipses\n",
    "sigmas = [3, 2, 1]\n",
    "colors = [\"#cfe8ff\", \"#7fb6ff\", \"#2c7be5\"]\n",
    "\n",
    "for nsig, color in zip(sigmas, colors):\n",
    "\n",
    "    width = 2 * nsig * np.sqrt(eigvals[0])\n",
    "    height = 2 * nsig * np.sqrt(eigvals[1])\n",
    "\n",
    "    ell = Ellipse(\n",
    "        (bt, br),\n",
    "        width,\n",
    "        height,\n",
    "        angle=angle,\n",
    "        facecolor=color,\n",
    "        edgecolor=\"black\",\n",
    "        linewidth=1.5,\n",
    "        alpha=0.85,\n",
    "        zorder=2,\n",
    "    )\n",
    "\n",
    "    ax.add_patch(ell)\n",
    "\n",
    "# Principal directions\n",
    "major = eigvecs[:, 0]\n",
    "minor = eigvecs[:, 1]\n",
    "\n",
    "Lmaj = 3.5 * np.sqrt(eigvals[0])\n",
    "Lmin = 3.5 * np.sqrt(eigvals[1])\n",
    "\n",
    "ax.plot(\n",
    "    [bt - Lmaj * major[0], bt + Lmaj * major[0]],\n",
    "    [br - Lmaj * major[1], br + Lmaj * major[1]],\n",
    "    \"--\",\n",
    "    lw=2.5,\n",
    "    color=\"#d62728\",\n",
    "    label=\"Major axis\",\n",
    ")\n",
    "\n",
    "ax.plot(\n",
    "    [bt - Lmin * minor[0], bt + Lmin * minor[0]],\n",
