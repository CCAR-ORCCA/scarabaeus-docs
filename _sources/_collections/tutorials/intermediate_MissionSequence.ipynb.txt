    "# set up\n",
    "km, sec, kg = scb.Units.get_units(['km', 'sec', 'kg'])\n",
    "J2000 = scb.Frame('J2000')\n",
    "\n",
    "t0 = scb.SpiceManager.cal2et('2026 JAN 01 00:00:00.000')\n",
    "tf = scb.SpiceManager.cal2et('2026 JAN 02 00:00:00.000')\n",
    "epochs = scb.EpochArray(np.arange(t0, tf, 500), 'TDB')\n",
    "\n",
    "# bodies\n",
    "jupiter = scb.CelestialBody.from_constants('JUPITER')\n",
    "sc = scb.Spacecraft('TEST SC', 10000, tot_mass = scb.ArrayWUnits(1e3, kg))\n",
    "\n",
    "pos0 = scb.ArrayWFrame(scb.ArrayWUnits([25e6, 0, 0], km), J2000)\n",
    "vel0 = scb.ArrayWFrame(scb.ArrayWUnits([3, 0, 0], km/sec), J2000)\n",
    "\n",
    "state_vector = scb.StateArray(epoch  = scb.EpochArray(scb.ArrayWUnits(t0, sec), 'TDB'),\n",
    "                              origin = jupiter,\n",
    "                              state  = scb.StateDefinition()\n",
    "                                          .position(sc, pos0)\n",
    "                                          .velocity(sc, vel0))\n",
    "\n",
    "# propagator\n",
    "fm = scb.ForceModelTranslation(sc)\n",
    "prop = scb.Propagator(primary_body  = sc, \n",
    "                      state_vector  = state_vector, \n",
    "                      tspan         = epochs, \n",
    "                      force_models  = fm, \n",
    "                      propagate_STM = False)\n",
    "\n",
    "# mission sequence\n",
    "sequence = scb.MissionSequence('TEST_SEQUENCE')\n",
    "sequence.addLeg('LEG1', scb.ArrayWUnits(86400, sec), state_vector, prop, scb.ArrayWUnits(10, sec))\n",
    "\n",
    "mission = scb.ScenarioSetup(sequence)\n",
    "mission.propagate()\n",
    "\n",
    "# NOTE: cant specify file location so this will fail when it tries to write the file\n",
    "orbiter_traj = scb.Trajectory('TEST_TRAJ', state_array = mission.propagated_state_array)"
