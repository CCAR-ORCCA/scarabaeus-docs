    "\n",
    "Orbiter_drymass = scb.ArrayWUnits(1275.0, kg)\n",
    "Orbiter_fuelmass = scb.ArrayWUnits(135, kg)\n",
    "Orbiter_area = scb.ArrayWUnits(1e-06, km**2)\n",
    "Orbiter_cr_srp = scb.ArrayWUnits(1, None)\n",
    "sc_true = scb.Spacecraft(\n",
    "    \"TEST_SC\",\n",
    "    -1099,\n",
    "    Orbiter_drymass + Orbiter_fuelmass,\n",
    "    Orbiter_area,\n",
    "    Orbiter_cr_srp,\n",
    ")\n",
    "spacecraft = sc_true\n",
    "\n",
    "# Initialize frame, planet constants and origin\n",
    "frame = J2000\n",
    "origin = scb.CelestialBody.from_constants(\"SUN\")\n",
    "\n",
    "# Initialize epochs\n",
    "time_0 = scb.SpiceManager.jd2et(2461809.72995654 + 1 / 3)\n",
    "time_f = scb.SpiceManager.jd2et(2461809.72995654 + 4)\n",
    "period = scb.ArrayWUnits(time_f - time_0, sec)\n",
    "dt = 30 * 30 * 30\n",
    "epoch_array = scb.EpochArray(np.arange(time_0, time_f, dt), sys=\"TDB\")\n",
    "\n",
    "# Get initial position and velocity\n",
    "initial_position = scb.ArrayWUnits(\n",
    "    np.array([-1.1123095885148e08, 8.9094345479316e07, 3.8656500948069e07]), km\n",
    ")\n",
    "initial_velocity = scb.ArrayWUnits(\n",
    "    np.array([-20.6936999825159, -16.7800270812616, -6.6437327193572]), km / sec\n",
    ")\n",
    "\n",
    "# Apply perturbation if the flag is set\n",
    "initial_position = scb.ArrayWFrame(initial_position, frame)\n",
    "initial_velocity = scb.ArrayWFrame(initial_velocity, frame)\n",
    "\n",
    "# Initialize first leg\n",
    "leg1_name = \"First Leg\"\n",
    "leg1_period = period\n",
    "leg1_state_definition = [\n",
    "    (\n",
    "        \"position\",\n",
    "        3,\n",
    "        \"estimated\",\n",
    "        \"dynamic\",\n",
    "        spacecraft,\n",
    "        initial_position,\n",
    "    ),\n",
    "    (\n",
    "        \"velocity\",\n",
    "        3,\n",
    "        \"estimated\",\n",
    "        \"dynamic\",\n",
    "        spacecraft,\n",
    "        initial_velocity,\n",
    "    ),\n",
    "]\n",
    "leg1_model = scb.StateArray(\n",
    "    epoch=epoch_array[0],\n",
    "    origin=origin,\n",
    "    state=scb.StateDefinition.from_components(leg1_state_definition),\n",
    ")\n",
    "leg1_dt = scb.ArrayWUnits(dt, sec)\n",
    "\n",
    "# Initialize first node\n",
    "dv1_name = \"Impulsive Burn\"\n",
    "dv1 = scb.ArrayWUnits(0 * np.array([0.16384638, 0.9830783, 0.08192319]), km / sec)\n",
    "dv1 = scb.ArrayWFrame(dv1, frame)\n",
    "dv1_model = scb.ImpulsiveBurn(delta_v=dv1)\n",
    "\n",
    "# Initialize second leg\n",
    "leg2_name = \"Second Leg\"\n",
    "leg2_period = period\n",
    "leg2_state_definition = [\n",
    "    (\n",
    "        \"position\",\n",
    "        3,\n",
    "        \"estimated\",\n",
    "        \"dynamic\",\n",
    "        spacecraft,\n",
    "        scb.ArrayWFrame(np.zeros(3), km, frame),  # Dummy value\n",

    "    (\n",
    "        \"velocity\",\n",
    "        3,\n",
    "        \"estimated\",\n",
    "        \"dynamic\",\n",
    "        spacecraft,\n",
    "        scb.ArrayWFrame(np.zeros(3), km / sec, frame),  # Dummy value\n",
    "    ),\n",
    "    (\n",
    "        \"dv_man\",\n",
    "        3,\n",
    "        \"estimated\",\n",
    "        \"static\",\n",
    "        spacecraft,\n",
    "        dv1,  # Perturbed value if flag is set\n",
    "    ),\n",
    "]\n",
    "leg2_model = scb.StateArray(\n",
    "    epoch=scb.EpochArray(np.zeros(1), sys=\"TDB\"),\n",

    "    state=scb.StateDefinition.from_components(leg2_state_definition),\n",
    ")\n",
    "leg2_dt = scb.ArrayWUnits(dt, sec)\n",
    "\n",
    "# Define propagator\n",
    "third_bodies = [\"MERCURY\", \"VENUS\", \"EARTH\"]\n",
